CREATE OR REPLACE FORCE VIEW NISA_SCHEMA.DATA_TRANSAKSI_V (ID_PENJUALAN, ID_BARANG, NAMA_BARANG, ID_PEMBELI, NAMA_PEMBELI, ID_TOKO, NAMA_TOKO, JUMLAH_BELI, TANGGAL_BELI, LOKASI_TOKO) AS 
  SELECT ID_PENJUALAN, 
	 ID_BARANG, 
	 NAMA_BARANG, 
	 ID_PEMBELI, 
	 NAMA_PEMBELI, 
	 ID_TOKO, 
	 NAMA_TOKO, 
	 JUMLAH_BELI, 
	 TANGGAL_BELI,
	 LOKASI_TOKO
     FROM (SELECT a.ID_PENJUALAN, a.ID_BARANG, b.NAMA, a.ID_PEMBELI, c.NAMA, a.ID_TOKO, d.NAMA_TOKO, a.JUMLAH_BELI, a.TANGGAL_BELI, a.LOKASI_TOKO
             FROM NISA_SCHEMA.DATA_PENJUALAN a
                  INNER JOIN NISA_SCHEMA.DATA_BARANG b
                     ON a.ID_BARANG = b.ID_BARANG    
                  INNER JOIN NISA_SCHEMA.DATA_PEMBELI c
                     ON a.ID_PEMBELI = c.ID_PEMBELI
                  INNER JOIN NISA_SCHEMA.TOKO d
                     ON a.ID_TOKO = d.ID_TOKO;
